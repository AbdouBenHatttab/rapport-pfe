% ============================================================
%  INTRODUCTION
% ============================================================
\ReportSection{Introduction}{Introduction}

{\fontsize{12}{16}\selectfont
\setlength{\parindent}{1.2cm}
\setlength{\parskip}{6pt}

\indent This chapter defines the product backlog planning for the proposed internal platform at ArabSoft. It identifies the actors, structures the functional and non-functional requirements, and formalizes user stories and backlog priorities that will guide sprint planning and implementation.

}

\vspace{6pt}

% ============================================================
%  I. ACTOR IDENTIFICATION
% ============================================================
\ReportSection{I. Actor Identification}{I. Actor Identification}

\ReportSubsection{I.1 Employee}{\hspace{0.5cm} 1. Employee}

{\fontsize{12}{18}\selectfont
\setlength{\parindent}{1.2cm}
\setlength{\parskip}{6pt}

\indent The employee submits administrative requests, tracks processing status, accesses assigned tasks, and receives notifications related to approvals, rejections, and project activities.

}

\vspace{6pt}

\ReportSubsection{I.2 Team Leader}{\hspace{0.5cm} 2. Team Leader}

{\fontsize{12}{18}\selectfont
\setlength{\parindent}{1.2cm}
\setlength{\parskip}{6pt}

\indent The team leader supervises team activities, validates requests requiring managerial approval, assigns and follows tasks, and monitors project execution indicators.

}

\vspace{6pt}

\ReportSubsection{I.3 HR Manager}{\hspace{0.5cm} 3. HR Manager}

{\fontsize{12}{18}\selectfont
\setlength{\parindent}{1.2cm}
\setlength{\parskip}{6pt}

\indent The HR manager manages employee records and roles, performs final validation of administrative requests, and monitors compliance through dashboards and reports.

}

\vspace{6pt}

\ReportSubsection{I.4 New User}{\hspace{0.5cm} 4. New User}

{\fontsize{12}{18}\selectfont
\setlength{\parindent}{1.2cm}
\setlength{\parskip}{6pt}

\indent A new user is registered in the platform but remains in a restricted state until HR validation and role assignment are completed.

}

\vspace{6pt}

% ============================================================
%  II. FUNCTIONAL REQUIREMENTS
% ============================================================
\ReportSection{II. Functional Requirements}{II. Functional Requirements}

\ReportSubsection{II.1 Authentication and access management}{\hspace{0.5cm} 1. Authentication and access management}

{\fontsize{12}{18}\selectfont
\setlength{\parindent}{1.2cm}
\setlength{\parskip}{6pt}

\indent The system shall support registration, secure authentication, logout, account activation by HR, and role-based access control according to user profile.

}

\vspace{6pt}

\ReportSubsection{II.2 Administrative request management}{\hspace{0.5cm} 2. Administrative request management}

{\fontsize{12}{18}\selectfont
\setlength{\parindent}{1.2cm}
\setlength{\parskip}{6pt}

\indent The platform shall manage submission, validation, rejection, and history tracking for leave requests, document requests, loan requests, and authorization requests.

}

\vspace{6pt}

\ReportSubsection{II.3 Team and employee management}{\hspace{0.5cm} 3. Team and employee management}

{\fontsize{12}{18}\selectfont
\setlength{\parindent}{1.2cm}
\setlength{\parskip}{6pt}

\indent The system shall provide profile management, team assignment, leader designation, and maintenance of employee administrative information.

}

\vspace{6pt}

\ReportSubsection{II.4 Project and task management}{\hspace{0.5cm} 4. Project and task management}

{\fontsize{12}{18}\selectfont
\setlength{\parindent}{1.2cm}
\setlength{\parskip}{6pt}

\indent The platform shall support project creation, task assignment, workload organization, progress tracking, and status visibility for all authorized stakeholders.

}

\vspace{6pt}

\ReportSubsection{II.5 Notification and reporting}{\hspace{0.5cm} 5. Notification and reporting}

{\fontsize{12}{18}\selectfont
\setlength{\parindent}{1.2cm}
\setlength{\parskip}{6pt}

\indent The system shall send operational notifications and provide reports and dashboards to support monitoring, traceability, and decision-making.

}

\vspace{6pt}

\ReportSubsection{II.6 AI assistance and decision support}{\hspace{0.5cm} 6. AI assistance and decision support}

{\fontsize{12}{18}\selectfont
\setlength{\parindent}{1.2cm}
\setlength{\parskip}{6pt}

\indent The platform shall provide AI-assisted support to help employees prepare administrative requests, assist team leaders in reviewing requests and workload constraints, and support HR managers with recommendation-oriented decision assistance while preserving human validation, traceability, and auditability.

}

\vspace{6pt}

% ============================================================
%  III. NON-FUNCTIONAL REQUIREMENTS
% ============================================================
\ReportSection{III. Non-Functional Requirements}{III. Non-Functional Requirements}

\ReportSubsection{III.1 Security}{\hspace{0.5cm} 1. Security}

{\fontsize{12}{18}\selectfont
\setlength{\parindent}{1.2cm}
\setlength{\parskip}{6pt}

\indent The application must guarantee secure authentication, protection of sensitive data, strict role-based permissions, and complete audit traceability.

}

\vspace{6pt}

\ReportSubsection{III.2 Performance}{\hspace{0.5cm} 2. Performance}

{\fontsize{12}{18}\selectfont
\setlength{\parindent}{1.2cm}
\setlength{\parskip}{6pt}

\indent The platform must maintain acceptable response times under expected internal usage and support concurrent request processing without significant delays.

}

\vspace{6pt}

\ReportSubsection{III.3 Usability}{\hspace{0.5cm} 3. Usability}

{\fontsize{12}{18}\selectfont
\setlength{\parindent}{1.2cm}
\setlength{\parskip}{6pt}

\indent Interfaces must be clear and intuitive to reduce training effort and facilitate adoption by employees, team leaders, and HR managers.

}

\vspace{6pt}

\ReportSubsection{III.4 Maintainability}{\hspace{0.5cm} 4. Maintainability}

{\fontsize{12}{18}\selectfont
\setlength{\parindent}{1.2cm}
\setlength{\parskip}{6pt}

\indent The system architecture must allow efficient correction, enhancement, and extension through modular and well-documented components.

}

\vspace{6pt}

\ReportSubsection{III.5 Scalability}{\hspace{0.5cm} 5. Scalability}

{\fontsize{12}{18}\selectfont
\setlength{\parindent}{1.2cm}
\setlength{\parskip}{6pt}

\indent The platform should support future growth in users, workflows, and modules without requiring major redesign.

}

\vspace{6pt}
